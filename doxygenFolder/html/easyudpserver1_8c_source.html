47 int ore_listen( int port )
48 {
49  int sd; // 待ち受けソケット
50  struct addrinfo hints; // bind用ヒント
51  struct addrinfo *ai; // bind用アドレス情報
52  char portstr[16];
53  int opt;
54  int result;
55 
56 
57  // listenポートの情報を設定
58  memset(&hints,0,sizeof(hints));
59  hints.ai_flags = AI_PASSIVE | AI_ADDRCONFIG; // 待ち受けの指定
60  hints.ai_family = AF_UNSPEC; // IPv6/IPv4 両方
61  hints.ai_socktype = SOCK_DGRAM; // UDP です
62  snprintf( portstr, sizeof(portstr), "%d", port ); // 待ち受けポート番号を文字列にする
63  result = getaddrinfo( NULL, portstr, &hints, &ai );
64  if( result!=0 ){
65  fprintf( stderr,"getaddrinfo: %s\n", gai_strerror(result) );
66  return -1;
67  }
68 
69  // 待ち受けソケット作成
70  sd = socket( ai->ai_family, ai->ai_socktype, ai->ai_protocol );
71  if( sd<0 ){
72  perror("socket");
73  freeaddrinfo(ai);
74  return -1;
75  }
76 
77  // ソケットオプション登録
78  opt = 1;
79  setsockopt( sd, SOL_SOCKET, SO_REUSEADDR, &opt, sizeof(opt) );
80 
81  // バインド
82  result = bind( sd, ai->ai_addr, ai->ai_addrlen );
83  if( result<0 ){
84  perror("bind");
85  close(sd);
86  freeaddrinfo(ai);
87  return -1;
88  }
89 
90  // 用が済んだのでアドレス情報を開放
91  freeaddrinfo(ai);
92 
93  return sd;
94 }
